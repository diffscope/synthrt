project(dsinfer
    LANGUAGES CXX
    VERSION 0.0.1.0
    DESCRIPTION "DiffSinger Inference Framework"
)

# Build Options
# ----------------------------------
option(DSINFER_ENABLE_DIRECTML "Enable DirectML provider" OFF)
